Tel het aantal gid (bv. is 1) van de afgelopen maand. Dat is de counter om te zien hoe vaak de groep of bedrijf wordt gebruikt. Daarbij sorteren gevolgd door de naam.
Opschonen door alles ouder dan een maand te verwijderen.

Counter tabel opschonen na het inloggen. Verwijder alles ouder dan 1 a 2 maanden.

Sorteer query. Laat de meest opgevraagde groep eerst zien en daarna sorteren op naam.

SELECT tbl_groups.`group`, count(0) AS ranking 
FROM `tbl_groups`
LEFT JOIN tbl_rankings ON tbl_groups.`id` = tbl_rankings.`gid`
GROUP BY tbl_groups.`group`
ORDER BY ranking DESC, `group`;

Crypto Sheet
------------

Services vanuit de tbl_wallets tabel:

SELECT tbl_services.`id` AS id,`service`
FROM tbl_wallets
LEFT JOIN tbl_accounts ON tbl_wallets.`aid` = tbl_accounts.`id`
LEFT JOIN tbl_services ON tbl_accounts.`sid` = tbl_services.`id`
LEFT JOIN tbl_cryptocurrenties ON tbl_wallets.`cid` = tbl_cryptocurrenties.`id`
WHERE tbl_wallets.`hide` = 0
GROUP BY id
ORDER BY `service`

Accounts vanuit de tbl_wallets tabel:

SELECT tbl_accounts.`id` AS id,`account`
FROM tbl_wallets
LEFT JOIN tbl_accounts ON tbl_wallets.`aid` = tbl_accounts.`id`
LEFT JOIN tbl_services ON tbl_accounts.`sid` = tbl_services.`id`
LEFT JOIN tbl_cryptocurrenties ON tbl_wallets.`cid` = tbl_cryptocurrenties.`id`
WHERE tbl_wallets.`hide` = 0 AND tbl_services.`id` = 16
GROUP BY id
ORDER BY `account`

Crypto symbol vanuit de tbl_wallets tabel:

SELECT tbl_cryptocurrenties.`id` AS id,`symbol`
FROM tbl_wallets
LEFT JOIN tbl_accounts ON tbl_wallets.`aid` = tbl_accounts.`id`
LEFT JOIN tbl_services ON tbl_accounts.`sid` = tbl_services.`id`
LEFT JOIN tbl_cryptocurrenties ON tbl_wallets.`cid` = tbl_cryptocurrenties.`id`
WHERE tbl_wallets.`hide` = 0 AND tbl_accounts.`id` = 6
GROUP BY id
ORDER BY `symbol`

-- Totale waarde per type (alle accounts).
SELECT 
	CASE 
		WHEN `type` = 'finance' then 'Financiën'
		WHEN `type` = 'stock'   then 'Aandelen'
		WHEN `type` = 'savings' then 'Spaargeld'
	END AS type,
	'all' AS `account`,       
	ROUND(SUM(value) * 100 / (SELECT SUM(value) AS v FROM tbl_value_accounts WHERE tbl_value_accounts .`date` = '2024-04-22'),2) AS `ratio`,
    SUM(`value`) AS `value`
FROM tbl_value_accounts 
INNER JOIN tbl_accounts ON tbl_value_accounts.`aid` = tbl_accounts.`id` 
WHERE tbl_value_accounts .`date` = '2024-04-22'
GROUP BY type
ORDER BY FIELD(type, 'finance', 'stock', 'savings');

-- Waarde per account.
SELECT 
	CASE 
		WHEN `type` = 'finance' then 'Financiën'
		WHEN `type` = 'stock'   then 'Aandelen'
		WHEN `type` = 'savings' then 'Spaargeld'
	END AS type,
	CAST(AES_DECRYPT(tbl_accounts.`account`,'Put your encryption key or phrase here!') AS CHAR(45)) AS account, 
    ROUND(value * 100 / (SELECT SUM(value) AS v FROM tbl_value_accounts WHERE tbl_value_accounts .`date` = '2024-04-22'),2) AS `ratio`, value
FROM tbl_value_accounts 
INNER JOIN tbl_accounts ON tbl_value_accounts.`aid` = tbl_accounts.`id` 
WHERE tbl_value_accounts .`date` = '2024-04-22'
ORDER BY FIELD(type, 'finance', 'stock', 'savings'), account;

-- Totale waarde per type (alle accounts).
SELECT 	
	CASE 
		WHEN `type` = 'finance' then 'Financiën'
		WHEN `type` = 'stock'   then 'Aandelen'
		WHEN `type` = 'savings' then 'Spaargeld'
		WHEN `type` = 'crypto'  then 'Crypto'
	END AS type, 'all' AS `service`, 'all' AS `account`, ROUND(100 * `value` / SUM(value) OVER (),2) ratio, SUM(`value`) AS `value`
FROM (
	SELECT type, SUM(`value`) AS `value`
	FROM tbl_value_accounts 
	INNER JOIN tbl_accounts ON tbl_value_accounts.`aid` = tbl_accounts.`id` 
	WHERE tbl_value_accounts .`date` = '2024-07-22'
	GROUP BY type
	UNION
	SELECT type, ROUND(SUM(amount*value),2) AS `value`
	FROM tbl_value_cryptos
	LEFT JOIN tbl_amount_wallets ON tbl_value_cryptos.`id` = tbl_amount_wallets.`vid`
	LEFT JOIN tbl_wallets ON tbl_amount_wallets.`wid` = tbl_wallets.`id`
	LEFT JOIN tbl_accounts ON tbl_wallets.`aid` = tbl_accounts.`id`
	WHERE tbl_value_cryptos.`date` = '2024-07-22'
) total
GROUP BY type
ORDER BY FIELD(type, 'finance', 'stock', 'savings', 'crypto');
